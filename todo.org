  #+begin_src elisp
    (defun insert-text-from-file (file-path)
    "Insert the contents of a text file at FILE-PATH into the minibuffer."
    (interactive "fEnter path to text file: ")
    (when (file-readable-p file-path)
      (with-temp-buffer
        (insert-file-contents file-path)
        (let ((text (buffer-string)))
          (minibuffer-message "Text from file: %s" text)))))

    (global-set-key (kbd "C-c i") 'insert-text-from-file)
  #+end_src
